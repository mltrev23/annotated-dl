Initial 1×1 convolution

Transformer layers

46        )

Final 1×1 convolution 

49        self.proj_out = nn.Conv2d(channels, channels, kernel_size=1, stride=1, padding=0)

#

Get shape [batch_size, channels, height, width] 

57        b, c, h, w = x.shape
#

For residual connection 

59        x_in = x

Normalize

Initial 1×1 convolution 

63        x = self.proj_in(x)

79class BasicTransformerBlock(nn.Module):
#
  • d_model is the input embedding size
  • n_heads is the number of attention heads
  • d_head is the size of a attention head
  • d_cond is the size of the conditional embeddings
83    def __init__(self, d_model: int, n_heads: int, d_head: int, d_cond: int):
# 
90        super().__init__()
#

Self-attention layer and pre-norm layer 

92        self.attn1 = CrossAttention(d_model, d_model, n_heads, d_head)
93        self.norm1 = nn.LayerNorm(d_model)
#

Cross attention layer and pre-norm layer 

95        self.attn2 = CrossAttention(d_model, d_cond, n_heads, d_head)
96        self.norm2 = nn.LayerNorm(d_model)
#

Feed-forward network and pre-norm layer 

98        self.ff = FeedForward(d_model)
99        self.norm3 = nn.LayerNorm(d_model)

  • x are the input embeddings of shape [batch_size, height * width, d_model]

101    def forward(self, x: torch.Tensor, cond: torch.Tensor):
#

Self attention 

107        x = self.attn1(self.norm1(x)) + x
#

Cross-attention with conditioning 

109        x = self.attn2(self.norm2(x), cond=cond) + x

Feed-forward network

Cross Attention Layer

This falls-back to self-attention when conditional embeddings are not specified.

116class CrossAttention(nn.Module):
#
  • d_model is the input embedding size
  • n_heads is the number of attention heads
  • d_head is the size of a attention head
  • d_cond is the size of the conditional embeddings
  • is_inplace specifies whether to perform the attention softmax computation inplace to save memory
122    def __init__(self, d_model: int, d_cond: int, n_heads: int, d_head: int, is_inplace: bool = True):
# 
131        super().__init__()
132
133        self.is_inplace = is_inplace
134        self.n_heads = n_heads
#

Attention scaling factor 

137        self.scale = d_head ** -0.5
#

Query, key and value mappings 

140        d_attn = d_head * n_heads
141        self.to_q = nn.Linear(d_model, d_attn, bias=False)
142        self.to_k = nn.Linear(d_cond, d_attn, bias=False)
143        self.to_v = nn.Linear(d_cond, d_attn, bias=False)
#

Final linear layer 

146        self.to_out = nn.Sequential(nn.Linear(d_attn, d_model))
#
  • x are the input embeddings of shape [batch_size, height * width, d_model]
  • cond is the conditional embeddings of shape [batch_size, n_cond, d_cond] 
148    def forward(self, x: torch.Tensor, cond: Optional[torch.Tensor] = None):
#

If cond is None we perform self attention 

155        if cond is None:
156            cond = x
#

Get query, key and value vectors 

159        q = self.to_q(x)
160        k = self.to_k(cond)
161        v = self.to_v(cond)
#

Split them to heads of shape [batch_size, seq_len, n_heads, d_head] 

164        q = q.view(*q.shape[:2], self.n_heads, -1)
165        k = k.view(*k.shape[:2], self.n_heads, -1)
166        v = v.view(*v.shape[:2], self.n_heads, -1)
#

Calculate attention dkey​​QK⊤​ 

169        attn = torch.einsum('bihd,bjhd->bhij', q, k) * self.scale
#

Compute softmax seqsoftmax​(dkey​​QK⊤​) 

173        if self.is_inplace:
174            half = attn.shape[0] // 2
175            attn[half:] = attn[half:].softmax(dim=-1)
176            attn[:half] = attn[:half].softmax(dim=-1)
177        else:
178            attn = attn.softmax(dim=-1)
#

Compute attention output seqsoftmax​(dkey​​QK⊤​)V 

182        out = torch.einsum('bhij,bjhd->bihd', attn, v)
#

Reshape to [batch_size, height * width, n_heads * d_head] 

184        out = out.reshape(*out.shape[:2], -1)
#

Map to [batch_size, height * width, d_model] with a linear layer 

186        return self.to_out(out)
