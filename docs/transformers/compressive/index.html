55    def with_memory(self, z: torch.Tensor, mem: Optional[torch.Tensor], c_mem: Optional[torch.Tensor]):
56        if mem is None:
57            return z
58
59        if c_mem is not None:
60            mem = torch.cat((c_mem, mem), dim=0)
61
62        mem = self.norm_self_attn(mem)
63        return torch.cat((mem, z), dim=0)
#
  • x are the token level feature vectors of shape [seq_len, batch_size, d_model]
  • mem are the past token level feature vectors of shape [mem_len + c_mem_len * c, batch_size, d_model]
  • mask is a matrix of shape [seq_len, c_mem_len + mem_len + seq_len, batch_size] or [seq_len, c_mem_len + mem_len + seq_len, 1]. mask[i, j] is true if token at i can see token at j.
65    def forward(self, *,
66                x: torch.Tensor,
67                mem: Optional[torch.Tensor],
68                c_mem: Optional[torch.Tensor],
69                mask: torch.Tensor):
#

Normalize the vectors before doing self attention

78        z = self.norm_self_attn(x)
79        m_z = self.with_memory(z, mem, c_mem)
#

Attention

81        self_attn = self.self_attn(query=z, key=m_z, value=m_z, mask=mask)

Add the attention results

83        x = x + self.dropout(self_attn)

133class AttentionReconstructionLoss:
# 
134    def __init__(self, layers: TypedModuleList[CompressiveTransformerLayer]):
135        self.layers = layers
136        self.loss_func = nn.MSELoss()
# 
138    def prepare_for_attn(self, pmha: PrepareForMultiHeadAttention, x: torch.Tensor):
139        head_shape = x.shape[:-1]
#

Linear transform

142        weight = pmha.linear.weight.detach()
143        bias = pmha.linear.bias.detach() if pmha.linear.bias is not None else None
144        x = F.linear(x, weight, bias)
#

Split last dimension into heads

147        x = x.view(*head_shape, pmha.heads, pmha.d_k)
#

Output has shape [seq_len, batch_size, heads, d_k] or [batch_size, d_model]

150        return x
# 
152    def attn(self, layer: RelativeMultiHeadAttention, query: torch.Tensor, key: torch.Tensor, value: torch.Tensor):
153        query = self.prepare_for_attn(layer.query, query)
154        key = self.prepare_for_attn(layer.key, key)
155        value = self.prepare_for_attn(layer.value, value)
#

Compute attention scores $Q K^\top$. This gives a tensor of shape [seq_len, seq_len, batch_size, heads].

159        scores = torch.einsum('ibhd,jbhd->ijbh', query, key)
#

Scale scores $\frac{Q K^\top}{\sqrt{d_k}}$

162        scores *= layer.scale
#

$softmax$ attention along the key sequence dimension $\underset{seq}{softmax}\Bigg(\frac{Q K^\top}{\sqrt{d_k}}\Bigg)$

166        attn = layer.softmax(scores)
#

Multiply by values 

170        return torch.einsum("ijbh,jbhd->ibhd", attn, value)
# 
172    def norm(self, ln: nn.LayerNorm, x: torch.Tensor):
173        weight = ln.weight.detach() if ln.weight is not None else None
174        bias = ln.bias.detach() if ln.bias is not None else None
175
176        return F.layer_norm(x, ln.normalized_shape, weight, bias, ln.eps)
# 
178    def calc_loss(self, layer: CompressiveTransformerLayer, h: torch.Tensor, mem: torch.Tensor):
179        h = h.detach()
180        mem = mem.detach()
181
182        c_mem = layer.compress(mem)
183
184        h = self.norm(layer.norm_self_attn, h)
185        mem = self.norm(layer.norm_self_attn, mem)
186        c_mem = self.norm(layer.norm_self_attn, c_mem)
187
188        return self.loss_func(self.attn(layer.self_attn, h, mem, mem),
189                              self.attn(layer.self_attn, h, c_mem, c_mem))
# 
191    def __call__(self, h: List[torch.Tensor], mem: List[torch.Tensor]):
192        losses = [self.calc_loss(layer, h[n], mem[n]) for n, layer in enumerate(self.layers)]
193        return sum(losses)
