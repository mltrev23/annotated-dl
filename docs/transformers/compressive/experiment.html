32class AutoregressiveModel(Module):
# 
37    def __init__(self, n_vocab: int, d_model: int, transformer: CompressiveTransformer):
38        super().__init__()
#

Token embedding module

40        self.src_embed = nn.Embedding(n_vocab, d_model)
#

Transformer

42        self.transformer = transformer
#

Final layer

44        self.generator = nn.Linear(d_model, n_vocab)
#

Masks

46        self.mask_x = None
47        self.mask_mem = None
# 
49    def forward(self, x: torch.Tensor, mem: CompressedMemory):
#

Length of the memory

51        if mem is not None:
52            mem, c_mem = mem.mem, mem.c_mem
53        else:
54            mem = []
55            c_mem = []
56
57        m_len = len(mem[0]) if mem else 0
58        if c_mem:
59            m_len += len(c_mem[0])
#

Create a subsequent mask for tokens

62        if self.mask_x is None or self.mask_x.shape[0] < len(x):
63            from labml_nn.transformers.utils import subsequent_mask
64            self.mask_x = subsequent_mask(len(x)).to(x.device)
#

Create an all ones (full visibility) mask for memory

66        if self.mask_mem is None or self.mask_mem.shape[1] < m_len or self.mask_mem.shape[0] < len(x):
67            self.mask_mem = self.mask_x.new_ones(len(x), m_len, 1)
#

Concatenate the masks if there is memory

70        if m_len:
71            mask = torch.cat((self.mask_mem[:len(x), :m_len], self.mask_x[:len(x), :len(x)]), dim=1)
#

Use the subsequent mask otherwise

73        else:
74            mask = self.mask_x[:len(x), :len(x)]
#

Token embeddings

77        x = self.src_embed(x)
#

Run it through the transformer

79        res, mem = self.transformer(x, mem, c_mem, mask)
#

Generate logits of the next token

81        res = self.generator(res)
# 
83        return res, mem

Concatenate memories and remove old memories to keep a maximum of mem_len memories.

126    @torch.no_grad()
127    def merge_memory(self, mem: CompressedMemory, new_mem: List[torch.Tensor]) \
128            -> Tuple[CompressedMemory, List[torch.Tensor]]:
#

If it’s configured not to use memory

135        if self.mem_len == 0:
136            return CompressedMemory([], []), []
137
138        if mem is not None:
139            mem, c_mem = mem.mem, mem.c_mem
140        else:
141            mem, c_mem = [], []
#

Concatenate with old memory

143        if mem:
144            mem = [torch.cat((m, x), dim=0) for m, x in zip(mem, new_mem)]
145        else:
146            mem = new_mem
147
148        if len(mem[0]) > self.mem_len:
149            n_c_mem = (len(mem[0]) - self.mem_len + self.compression_ratio - 1) // self.compression_ratio
150            old_mem = []
151            trunc_mem = []
152            for m in mem:
153                n_old = n_c_mem * self.compression_ratio
154                cm, m = torch.split(m, [n_old, len(m) - n_old])
155                old_mem.append(cm)
156                trunc_mem.append(m)
157            mem = trunc_mem
158
159            new_c_mem = []
160            for i, layer in enumerate(self.model.transformer.layers):
161                new_c_mem.append(layer.compress(old_mem[i]))
162
163            if c_mem:
164                c_mem = [torch.cat((m, nm), dim=0) for m, nm in zip(c_mem, new_c_mem)]
165            else:
166                c_mem = new_c_mem
#

Truncate old memories

169            if len(c_mem[0]) > self.c_mem_len:
170                c_mem = [m[-self.c_mem_len:] for m in c_mem]
171        else:
172            old_mem = []
# 
175        return CompressedMemory(mem, c_mem), old_mem
#

Training/validation step

177    def step(self, batch: any, batch_idx: BatchIndex):
#

Move data to the device

183        data, target = batch[0].to(self.device), batch[1].to(self.device)
#

Update global step (number of tokens processed) when in training mode

186        if self.mode.is_train:
187            tracker.add_global_step(data.shape[0] * data.shape[1])
#

Whether to capture model outputs

190        with self.mode.update(is_log_activations=batch_idx.is_last):
#

Get memories

192            mem = self.memory.get()
#

Run the model

194            output, new_mem = self.model(data, mem)
#

Merge memory

196            mem, old_mem = self.merge_memory(mem, new_mem)
#

Update memories

198            self.memory.set(mem)
#

Calculate and log cross entropy loss

201        loss = self.loss_func(output, target)
202        tracker.add("loss.", loss)
203
204        if old_mem:
205            ar_loss = self.attention_reconstruction_loss(new_mem, old_mem)
206            tracker.add("ar_loss.", ar_loss)
207            loss = loss + ar_loss
#

Calculate and log accuracy

210        self.accuracy(output, target)
211        self.accuracy.track()
#

Train the model

214        if self.mode.is_train:
#

Calculate gradients

216            loss.backward()
#

Clip gradients

218            torch.nn.utils.clip_grad_norm_(self.model.parameters(), max_norm=self.grad_norm_clip)
#

Take optimizer step

220            self.optimizer.step()
#

Log the model parameters and gradients on last batch of every epoch

222            if batch_idx.is_last:
223                tracker.add('model', self.model)
#

Clear the gradients

225            self.optimizer.zero_grad()
#

Save the tracked metrics

228        tracker.save()

230    def sample(self):
#

Starting prompt

236        prompt = self.prompt
#

Collect output for printing

238        log = [(prompt, Text.subtle)]
#

memory

240        mem = CompressedMemory([], [])
#

Sample 25 tokens

242        for i in monit.iterate('Sample', 25):
#

Tokenize the prompt

244            data = self.text.text_to_i(prompt).unsqueeze(-1)
#

Move to device

246            data = data.to(self.device)
#

Get the model output

248            output, new_mem = self.model(data, mem)
#

Get the model prediction (greedy)

250            output = output.argmax(dim=-1).squeeze(1)
#

Add the prediction to prompt

252            prompt += self.prompt_separator + self.text.itos[output[-1]]
#

Only feed the last character to model in next iteration, rest will go in as memories

254            prompt = prompt[-1:]
#

Add the prediction for logging

256            log += [(self.prompt_separator + self.text.itos[output[-1]], Text.value)]
#

Update memory

258            mem, _ = self.merge_memory(mem, new_mem)
#

Print the sampled output

261        logger.log(log)

264@option(Configs.model)
265def autoregressive_model(c: Configs):
# 
269    from labml_nn.transformers.xl import RelativeMultiHeadAttention
270    from labml_nn.transformers.feed_forward import FeedForward
271    m = AutoregressiveModel(c.n_tokens, c.d_model, CompressiveTransformer(
272        CompressiveTransformerLayer(d_model=c.d_model,
273                                    self_attn=RelativeMultiHeadAttention(c.heads, c.d_model, c.dropout),
274                                    feed_forward=FeedForward(c.d_model, c.d_ff, c.dropout),
275                                    dropout_prob=c.dropout,
276                                    compress=Conv1dCompression(c.compression_ratio, c.d_model)), c.n_layers))
277    return m.to(c.device)
