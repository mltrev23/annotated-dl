17                 dropout_prob: float):

24        super().__init__()
25        self.size = d_model

Whether to save input to the feed forward layer

32        self.is_save_ff_input = False

34    def forward(self, *,
35                x: torch.Tensor,
36                mem: Optional[torch.Tensor],
37                mask: torch.Tensor):
#

Normalize the vectors before doing self attention

39        z = self.norm_self_attn(x)
#

Normalize memory

41        if mem is not None:
42            mem = self.norm_self_attn(mem)
43            zm = torch.cat((mem, z), dim=0)
44        else:
45            zm = z
#

Run through self attention, i.e. keys and values are from self

47        self_attn = self.self_attn(query=z, key=zm, value=zm, mask=mask)
#

Add the self attention results

49        x = x + self.dropout(self_attn)
#

Normalize for feed-forward

52        z = self.norm_ff(x)
#

Save the input to the feed forward layer if specified

54        if self.is_save_ff_input:
55            self.ff_input = z.clone()
#

Pass through the feed-forward network

57        ff = self.feed_forward(z)
#

Add the feed-forward results back

59        x = x + self.dropout(ff)
60
61        return x

Final normalization layer

70        self.norm = nn.LayerNorm([layer.size])

72    def forward(self, x: torch.Tensor, mem: List[torch.Tensor], mask: torch.Tensor):
73        new_mem = []

Run through each transformer layer

75        for i, layer in enumerate(self.layers):

77            m = mem[i] if mem else None

80        return self.norm(x), new_mem
