Fast weights transformer

This paper compares linear self attention to fast weight systems and makes modifications to self attention update rule based on that. It also introduces a simpler, yet effective kernel function.

Fast weights

Consider a sequence of inputs $\big\{x^{(i)}\big\}^L_{i=1}$ or length $L$ and each step is a vector of size $d_{in}$; i.e. $x \in \mathbb{R}^{d_{in}}$. The fast weight model generates a weight matrix at each step to produce output $\big\{y^{(i)}\big\}^L_{i=1}$, $y \in \mathbb{R}^{d_{out}}$

$\otimes$ is the outer product ($a \otimes b = a b^\top$), where elements of the two vectors are multiplied with each other to give a matrix. $\sigma$ is an activation function. $\color{orange}{W_a}$ and $\color{orange}{W_b}$ are trainable weights (parameters). $\color{cyan}{W^{(i)}}$ are the fast weights that are generated at each step.

85class DPFP(Module):
# 
86    def __init__(self, nu: int = 1, eps: float = 1e-6):
87        super().__init__()
88        self.nu = nu
89        self.r = nn.ReLU()
90        self.eps = eps
# 
92    def __call__(self, x: torch.Tensor):
93        x = self.dpfp(x)
94        return x / (torch.sum(x, dim=-1, keepdim=True) + self.eps)
# 
96    def dpfp(self, x: torch.Tensor):
97        x = torch.cat([self.r(x), self.r(-x)], dim=-1)
98        x_rolled = torch.cat([x.roll(shifts=j, dims=-1) for j in range(1, self.nu + 1)], dim=-1)
99        x_repeat = torch.cat([x] * self.nu, dim=-1)
100
101        return x_repeat * x_rolled

104class FastWeightAttention(Module):
# 
105    def __init__(self, heads: int, d_model: int, dropout_prob: float, phi: DPFP):
106        super().__init__()
#

Number of features per head

109        self.d_k = d_model // heads
# 
111        self.heads = heads
#

These transform the query multi-headed attention.

114        self.query = PrepareForMultiHeadAttention(d_model, heads, self.d_k, bias=False)
#

These transform the key and value for multi-headed attention.

116        self.key = PrepareForMultiHeadAttention(d_model, heads, self.d_k, bias=False)
117        self.value = PrepareForMultiHeadAttention(d_model, heads, self.d_k, bias=False)
118
119        self.gate = nn.Sequential(PrepareForMultiHeadAttention(d_model, heads, 1, bias=False),
120                                  nn.Sigmoid())
121
122        self.phi = phi
#

Output layer

125        self.output = nn.Linear(d_model, d_model)
#

Dropout

127        self.dropout = nn.Dropout(dropout_prob)
# 
129    def __call__(self, x: torch.Tensor):
130        seq_len = x.shape[0]
131        query = self.phi(self.query(x))
132        key = self.phi(self.key(x))
133        value = self.value(x)
134        beta = self.gate(x)
135
136        weights = key.new_zeros((key.shape[1], key.shape[2], value.shape[3], key.shape[3]))
137        outputs = []
138
139        for i in range(seq_len):
140            value_existing = torch.einsum('bhvk,bhk->bhv', weights, key[i])
141
142            weights = weights + torch.einsum('bhv,bhk->bhvk', beta[i] * (value[i] - value_existing), key[i])
143
144            x = torch.einsum('bhvk,bhk->bhv', weights, query[i])
#

Concatenate multiple heads

147            outputs.append(x.reshape(x.shape[0], -1))
148
149        x = torch.stack(outputs)
#

Output layer

151        return self.output(x)

158                 feed_forward: FeedForward,
159                 dropout_prob: float):

Transformer size $d_{model}$

162        self.size = d_model
#

165        self.feed_forward = feed_forward
166        self.dropout = nn.Dropout(dropout_prob)
#

Normalization layers

169        self.norm_self_attn = nn.LayerNorm([d_model])
170        self.norm_ff = nn.LayerNorm([d_model])
# 
172    def __call__(self, x: torch.Tensor):
173        attn = self.attn(x)
#

Add the self attention results

175        x = x + self.dropout(attn)
#

Normalize for feed-forward

178        z = self.norm_ff(x)
#

Pass through the feed-forward network

180        ff = self.feed_forward(z)
#

Add the feed-forward results back

182        x = x + self.dropout(ff)
# 
185        return x
