23class CIFAR10VGGModel(Module):
# 
28    def conv_block(self, in_channels, out_channels) -> nn.Module:
29        return nn.Sequential(
30            nn.Conv2d(in_channels, out_channels, kernel_size=3, padding=1),
31            nn.ReLU(inplace=True),
32        )
# 
34    def __init__(self, blocks: List[List[int]]):
35        super().__init__()
#

5 $2 \times 2$ pooling layers will produce a output of size $1 \ times 1$. CIFAR 10 image size is $32 \times 32$

39        assert len(blocks) == 5
40        layers = []
#

RGB channels

42        in_channels = 3
#

Number of channels in each layer in each block

44        for block in blocks:
#

Convolution, Normalization and Activation layers

46            for channels in block:
47                layers += self.conv_block(in_channels, channels)
48                in_channels = channels
#

Max pooling at end of each block

50            layers += [nn.MaxPool2d(kernel_size=2, stride=2)]
#

Create a sequential model with the layers

53        self.layers = nn.Sequential(*layers)
#

Final logits layer

55        self.fc = nn.Linear(in_channels, 10)
# 
57    def __call__(self, x):
#

The VGG layers

59        x = self.layers(x)
#

Reshape for classification layer

61        x = x.view(x.shape[0], -1)
#

Final linear layer

63        return self.fc(x)
